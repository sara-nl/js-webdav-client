 26 /**
 27  * @class Connection to a WebDAV server
 28  *
 29  * @param  {String}   [host]            Optional; The hostname or IP address of the server. This is only needed if the host is different from the one serving this library.
 30  * @param  {Boolean}  [useHTTPS=false]  Optional; If set to true, HTTPS is used. If set to false or omitted, HTTP is used. This parameter is ignored if host is not set.
 31  * @param  {Number}   [port]            Optional; Set a custom port to connect to. If not set, the default port will be used (80 for HTTP and 443 for HTTPS). This parameter is ignored if host is not set.
 32  */
 33 nl.sara.webdav.Client = function(host, useHTTPS, port) {
 34   // First define private attributes
 35   Object.defineProperty(this, '_baseUrl', {
 36     'value': null,
 37     'enumerable': false,
 38     'configurable': false,
 39     'writable': true
 40   });
 41 
 42   // Constructor logic
 43   if (host !== undefined) {
 44     var protocol = (useHTTPS === true) ? 'https' : 'http';
 45     port = (port != undefined) ? port : ((protocol == 'https') ? 443 : 80);
 46     this._baseUrl = protocol + '://' + host + ((((protocol == 'http') && (port == 80)) || ((protocol == 'https') && (port == 443))) ? '' : ':' + port);
 47   }
 48 }

 62 /**
 63  * Converts a path to the full url (i.e. appends the protocol and host part)
 64  *
 65  * @param   {String}  path  The path on the server
 66  * @returns {String}        The full url to the path
 67  */
 68 nl.sara.webdav.Client.prototype.getUrl = function(path) {
 69   if (path.substring(0,1) != '/') {
 70     path = '/' + path;
 71   }
 72   if (this._baseUrl !== null) {
 73     return this._baseUrl + path;
 74   }else{
 75     return path;
 76   }
 77 }

 79 /**
 80  * Perform a WebDAV PROPFIND request
 81  *
 82  * @param    {String}                         path             The path get a PROPFIND for
 83  * @param    {Function(status,body,headers)}  callback         Querying the server is done asynchronously, this callback function is called when the results are in
 84  * @param    {String}                         [depth=0]        Optional; Value for the 'depth' header, should be either '0', '1' or the class constant INFINITY. When omitted, '0' is used. See RFC 4916.
 85  * @param    {mixed}                          [props=ALLPROP]  Optional; The properties to fetch. Should be either one of the class constants PROPNAME or ALLPROP or an array with Property objects. When omitted, ALLPROP is assumed. See RFC 4916.
 86  * @param    {nl.sara.webdav.Property[]}      [include]        Optional; An array with Property objects used for the <include> element. This is only used for ALLPROP requests. When omitted, no <include> element is send. See RFC 4916.
 87  * @param    {Array}                          headers          Optional; Additional headers to set
 88  * @returns  {nl.sara.webdav.Client}                           The client itself for chaining methods
 89  */
 90 nl.sara.webdav.Client.prototype.propfind = function(path, callback, depth, props, include, headers) {
 91   if ((path === undefined) || (callback === undefined)) {
 92     throw new nl.sara.webdav.Exception('PROPFIND requires the parameters path and callback', nl.sara.webdav.Exception.MISSING_REQUIRED_PARAMETER);
 93   }
 94   if (!(typeof path == "string")) {
 95     throw new nl.sara.webdav.Exception('PROPFIND parameter; path should be a string', nl.sara.webdav.Exception.WRONG_TYPE);
 96   }
 97 
 98   // Get the full URL, based on the specified path
 99   var url = this.getUrl(path);
100 
101   // Check the depth header
102   if (depth === undefined) { // We default depth to 0, not to infinity as this is not supported by all servers
103     depth = 0;
104   }
105   var depthHeader = null;
106   switch (depth) {
107     case 0:
108     case 1:
109       depthHeader = depth;
110       break;
111     case nl.sara.webdav.Client.INFINITY:
112       depthHeader = 'infinity';
113       break;
114     default:
115       throw new nl.sara.webdav.Exception("Depth header should be '0', '1' or nl.sara.webdav.Client.INFINITY", nl.sara.webdav.Exception.WRONG_VALUE);
116     break;
117   }
118 
119   // Create the request XML
120   if (props === undefined) {
121     props = nl.sara.webdav.Client.ALLPROP;
122   }
123   var propsBody = document.implementation.createDocument("DAV:", "propfind", null);
124   switch (props) { // Find out what the request is for
125     case nl.sara.webdav.Client.PROPNAME: // User wants all property names
126       propsBody.documentElement.appendChild(propsBody.createElementNS('DAV:', 'propname'));
127       break;
128     case nl.sara.webdav.Client.ALLPROP: // User wants all properties
129       propsBody.documentElement.appendChild(propsBody.createElementNS('DAV:', 'allprop'));
130       if (include !== undefined) { // There is content for the <DAV: include> tags, so parse it
131         if (!(include instanceof Array)) {
132           throw new nl.sara.webdav.Exception('Propfind parameter; include should be an array', nl.sara.webdav.Exception.WRONG_TYPE);
133         }
134         var includeBody = propsBody.createElementNS('DAV:', 'include');
135         for (var i = 0; i < include.length; i++) {
136           var item = include[i];
137           if (!nl.sara.webdav.Ie.isIE && !(item instanceof nl.sara.webdav.Property)) {
138             continue;
139           }
140           includeBody.appendChild(propsBody.createElementNS(item.namespace, item.tagname));
141         }
142         if (includeBody.hasChildNodes()) { // But only really add the <include> tag if there is valid content
143           propsBody.documentElement.appendChild(includeBody);
144         }
145       }
146       break;
147     default: // The default is to expect an array with Property objects; the user wants the values of these properties
148       if (!(props instanceof Array)) {
149         throw new nl.sara.webdav.Exception('Props parameter should be nl.sara.webdav.Client.PROPNAME, nl.sara.webdav.Client.ALLPROP or an array with Property objects', nl.sara.webdav.Exception.WRONG_VALUE);
150       }
151       var propertyBody = propsBody.createElementNS('DAV:', 'prop');
152       for (var i = 0; i < props.length; i++) { // Cycle through the array
153         var prop = props[i];
154         if (!nl.sara.webdav.Ie.isIE && !(prop instanceof nl.sara.webdav.Property)) {
155           continue;
156         }
157         propertyBody.appendChild(propsBody.createElementNS(prop.namespace, prop.tagname));
158       }
159       if (!propertyBody.hasChildNodes()) { // But if no properties are found, then the array didn't have Property objects in it
160         throw new nl.sara.webdav.Exception("Propfind parameter; if props is an array, it should be an array of Properties", nl.sara.webdav.Exception.WRONG_TYPE);
161       }
162       propsBody.documentElement.appendChild(propertyBody);
163     break;
164   }
165   var serializer = new XMLSerializer();
166   var body = '<?xml version="1.0" encoding="utf-8" ?>' + serializer.serializeToString(propsBody);
167 
168   // And then send the request
169   var ajax = nl.sara.webdav.Client.getAjax("PROPFIND", url, callback, headers);
170   ajax.setRequestHeader('Depth', depthHeader);
171   ajax.setRequestHeader('Content-Type', 'application/xml; charset="utf-8"');
172   ajax.send(body);
173 
174   return this;
175 }

177 /**
178  * Perform a WebDAV PROPPATCH request
179  *
180  * @param    {String}                         path        The path do a PROPPATCH on
181  * @param    {Function(status,body,headers)}  callback    Querying the server is done asynchronously, this callback function is called when the results are in
182  * @param    {nl.sara.webdav.Property[]}      [setProps]  Optional; The properties to set. If not set, delProps should be set. Can be omitted with 'undefined'.
183  * @param    {nl.sara.webdav.Property[]}      [delProps]  Optional; The properties to delete. If not set, setProps should be set.
184  * @param    {Array}                          headers     Optional; Additional headers to set
185  * @returns  {nl.sara.webdav.Client}                      The client itself for chaining methods
186  */
187 nl.sara.webdav.Client.prototype.proppatch = function(path, callback, setProps, delProps, headers) {
188   if ((path === undefined) || (callback === undefined) || ((setProps === undefined) && (delProps === undefined))) {
189     throw new nl.sara.webdav.Exception('PROPPATCH requires the parameters path, callback and at least one of setProps or delProps', nl.sara.webdav.Exception.MISSING_REQUIRED_PARAMETER);
190   }
191   if (!(typeof path == "string") || ((setProps !== undefined) && !(setProps instanceof Array)) || ((delProps !== undefined) && !(delProps instanceof Array))) {
192     throw new nl.sara.webdav.Exception('PROPPATCH parameter; path should be a string, setProps and delProps should be arrays', nl.sara.webdav.Exception.WRONG_TYPE);
193   }
194 
195   // Get the full URL, based on the specified path
196   var url = this.getUrl(path);
197 
198   // Create the request XML
199   var propsBody = document.implementation.createDocument("DAV:", "propertyupdate", null);
200   propsBody.documentElement.setAttribute("xmlns:D", "DAV:");
201   if (setProps !== undefined) {
202     var props = propsBody.createElementNS('DAV:', 'prop');
203     for (var i = 0; i < setProps.length; i++) { // Cycle through the array
204       var prop = setProps[i];
205       if (!nl.sara.webdav.Ie.isIE && !(prop instanceof nl.sara.webdav.Property)) {
206         continue;
207       }
208       var element = propsBody.createElementNS(prop.namespace, prop.tagname);
209       for (var j = 0; j < prop.xmlvalue.length; j++) {
210         var nodeCopy = propsBody.importNode(prop.xmlvalue.item(j), true);
211         element.appendChild(nodeCopy);
212       }
213       props.appendChild(element);
214     }
215     if (!props.hasChildNodes()) { // But if no properties are found, then the array didn't have Property objects in it
216       throw new nl.sara.webdav.Exception("Proppatch parameter; setProps should be an array of Properties", nl.sara.webdav.Exception.WRONG_TYPE);
217     }
218     var set = propsBody.createElementNS('DAV:', 'set');
219     set.appendChild(props);
220     propsBody.documentElement.appendChild(set);
221   }
222   if (delProps !== undefined) {
223     var props = propsBody.createElementNS('DAV:', 'prop');
224     for (var i = 0; i < delProps.length; i++) { // Cycle through the array
225       var prop = delProps[i];
226       if (!nl.sara.webdav.Ie.isIE && !(prop instanceof nl.sara.webdav.Property)) {
227         continue;
228       }
229       var element = propsBody.createElementNS(prop.namespace, prop.tagname);
230       props.appendChild(element);
231     }
232     if (!props.hasChildNodes()) { // But if no properties are found, then the array didn't have Property objects in it
233       throw new nl.sara.webdav.Exception("Proppatch parameter; delProps should be an array of Properties", nl.sara.webdav.Exception.WRONG_TYPE);
234     }
235     var remove = propsBody.createElementNS('DAV:', 'remove');
236     remove.appendChild(props);
237     propsBody.documentElement.appendChild(remove);
238   }
239   var serializer = new XMLSerializer();
240   var body = '<?xml version="1.0" encoding="utf-8" ?>' + serializer.serializeToString(propsBody);
241 
242   // And then send the request
243   var ajax = nl.sara.webdav.Client.getAjax("PROPPATCH", url, callback, headers);
244   ajax.setRequestHeader('Content-Type', 'application/xml; charset="utf-8"');
245   ajax.send(body);
246 
247   return this;
248 }

441 /**
442  * Perform a WebDAV COPY request
443  *
444  * @param    {String}                         path                              The path to perform COPY on
445  * @param    {Function(status,body,headers)}  callback                          Querying the server is done asynchronously, this callback function is called when the results are in
446  * @param    {String}                         destination                       The destination to copy to. Should be either a full URL or a path from the root on this server (i.e. it should start with a /)
447  * @param    {Boolean}                        [overwriteMode=SILENT_OVERWRITE]  Optional; Specify what to do when destination resource already exists. Should be either FAIL_ON_OVERWRITE or SILENT_OVERWRITE. The default is SILENT_OVERWRITE.
448  * @param    {String}                         [depth]                           Optional; Should be '0' or 'infinity'. This is used in case of a collection; 0 means only copy the collection itself, infinity means copy also everything contained in the collection
449  * @param    {Array}                          headers                           Optional; Additional headers to set
450  * @returns  {nl.sara.webdav.Client}                                            The client itself for chaining methods
451  */
452 nl.sara.webdav.Client.prototype.copy = function(path, callback, destination, overwriteMode, depth, headers) {
453   if ((path === undefined) || (callback === undefined) || (destination === undefined)) {
454     throw new nl.sara.webdav.Exception('COPY requires the parameters path, callback and destination', nl.sara.webdav.Exception.MISSING_REQUIRED_PARAMETER);
455   }
456   if ((typeof path != "string") || (typeof destination != "string")){
457     throw new nl.sara.webdav.Exception('COPY parameter; path and destination should be strings', nl.sara.webdav.Exception.WRONG_TYPE);
458   }
459 
460   // Get the full URL, based on the specified path
461   var url = this.getUrl(path);
462 
463   // If the destination starts with a / it is a absolute url on the same host, so prepare a complete URL
464   if (destination.substr(0,1) == '/') {
465     destination = this.getUrl(destination);
466   } // Else I assume it is a complete URL already
467 
468   // And then send the request
469   var ajax = nl.sara.webdav.Client.getAjax("COPY", url, callback, headers);
470   ajax.setRequestHeader('Destination', destination);
471   if (depth !== undefined) {
472     if ((depth != 0) && (depth != 'infinity')) {
473       throw new nl.sara.webdav.Exception("COPY parameter; depth should be '0' or 'infinity'", nl.sara.webdav.Exception.WRONG_VALUE);
474     }
475     ajax.setRequestHeader('Depth', depth);
476   }
477   if (overwriteMode == nl.sara.webdav.Client.FAIL_ON_OVERWRITE) {
478     ajax.setRequestHeader('Overwrite', 'F');
479   }
480   ajax.send();
481 
482   return this;
483 }

485 /**

488  * @param    {String}                         path                              The path to perform MOVE on
489  * @param    {Function(status,body,headers)}  callback                          Querying the server is done asynchronously, this callback function is called when the results are in
490  * @param    {String}                         destination                       The destination to move to. Should be either a full URL or a path from the root on this server (i.e. it should start with a /)
491  * @param    {Number}                         [overwriteMode=SILENT_OVERWRITE]  Optional; Specify what to do when destination resource already exists. Should be either FAIL_ON_OVERWRITE, TRUNCATE_ON_OVERWRITE or SILENT_OVERWRITE. The default is SILENT_OVERWRITE.
492  * @param    {Array}                          headers                           Optional; Additional headers to set
493  * @returns  {nl.sara.webdav.Client}                                            The client itself for chaining methods
494  */
495 nl.sara.webdav.Client.prototype.move = function(path, callback, destination, overwriteMode, headers) {
496   if ((path === undefined) || (callback === undefined) || (destination === undefined)) {
497     throw new nl.sara.webdav.Exception('MOVE requires the parameters path, callback and destination', nl.sara.webdav.Exception.MISSING_REQUIRED_PARAMETER);
498   }
499   if ((typeof path != "string") || (typeof destination != "string")){
500     throw new nl.sara.webdav.Exception('MOVE parameter; path and destination should be strings', nl.sara.webdav.Exception.WRONG_TYPE);
501   }
502 
503   // Get the full URL, based on the specified path
504   var url = this.getUrl(path);
505 
506   // If the destination starts with a / it is a absolute url on the same host, so prepare a complete URL
507   if (destination.substr(0,1) == '/') {
508     destination = this.getUrl(destination);
509   } // Else I assume it is a complete URL already
510 
511   // And then send the request
512   var ajax = nl.sara.webdav.Client.getAjax("MOVE", url, callback, headers);
513   ajax.setRequestHeader('Destination', destination);
514   if (overwriteMode == nl.sara.webdav.Client.FAIL_ON_OVERWRITE) {
515     ajax.setRequestHeader('Overwrite', 'F');
516   }else if (overwriteMode == nl.sara.webdav.Client.TRUNCATE_ON_OVERWRITE) {
517     ajax.setRequestHeader('Overwrite', 'T');
518   }
519   ajax.send();
520 
521   return this;
522 }

550 /**
551  * Prepares a XMLHttpRequest object to be used for an AJAX request
552  *
553  * @static
554  * @param    {String}                         method    The HTTP/webDAV method to use (e.g. GET, POST, PROPFIND)
555  * @param    {String}                         url       The url to connect to
556  * @param    {Function(status,body,headers)}  callback  Querying the server is done asynchronously, this callback function is called when the results are in
557  * @param    {Array}                          headers   Additional headers to set
558  * @returns  {XMLHttpRequest}                           A prepared XMLHttpRequest
559  */
560 nl.sara.webdav.Client.getAjax = function(method, url, callback, headers) {
561   var /** @type XMLHttpRequest */ ajax = new XMLHttpRequest();
562   ajax.open(method, url, true);
563   ajax.onreadystatechange=function(){nl.sara.webdav.Client.ajaxHandler(ajax, callback);}
564   return ajax;
565 }

567 /**
568  * AJAX request handler. Parses Multistatus (if available) and call a user specified callback function
569  *
570  * @static
571  * @param    {XMLHttpRequest}                 ajax      The XMLHttpRequest object which performed the request
572  * @param    {Function(status,body,headers)}  callback  Querying the server is done asynchronously, this callback function is called when the results are in
573  * @returns  {void}
574  */
575 nl.sara.webdav.Client.ajaxHandler = function(ajax, callback) {
576   if (ajax.readyState==4){ //if request has completed
577     var body = ajax.responseText;
578     if (ajax.status == 207) {
579       // Parse the response to a Multistatus object
580       for (var counter = 0; counter < ajax.responseXML.childNodes.length; counter++) {
581         if (nl.sara.webdav.Ie.getLocalName(ajax.responseXML.childNodes.item(counter)) == 'multistatus') {
582           body = new nl.sara.webdav.Multistatus(ajax.responseXML.childNodes.item(counter));
583           break;
584         }
585       }
586     }
587     callback(ajax.status, body, ajax.getAllResponseHeaders());
588   }
589 }
